Перечисления в Java - это специальный тип данных, который представляет собой фиксированный набор констант.
Вот пример объявления и использования перечисления в Java:
// Объявление перечисления
enum DayOfTheWeek {
    ПОНЕДЕЛЬНИК,
    ВТОРНИК,
    СРЕДА,
    ЧЕТВЕРГ,
    ПЯТНИЦА,
    СУББОТА,
    ВОСКРЕСЕНЬЕ
}

public class ПримерПеречисления {
    public static void main(String[] args) {
        // Использование перечисления
        DayOfTheWeek ToDay = DayOfTheWeek.ПОНЕДЕЛЬНИК;

        // Вывод значения перечисления
        System.out.println("Сегодня " + ToDay);

        // Сравнение значений перечисления
        if (ToDay == DayOfTheWeek.ПОНЕДЕЛЬНИК) {
            System.out.println("Сегодня понедельник");
        } else {
            System.out.println("Сегодня не понедельник");
        }
    }
}
